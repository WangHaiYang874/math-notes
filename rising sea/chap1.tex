    \begin{itemize}
        \item 一组对象\(\obj(\cate)\)
        \item 每对对象 \(A,B\) 间的态射集\(\mor(A,B)\ni f\). 
        \item 态射的复合: 对\(f : A\to B, g: B\to C\), 有\(h = g\circ f \in \mor(A\to C)\)

    \end{itemize}
\end{defi}
